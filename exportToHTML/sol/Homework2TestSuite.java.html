    @Before

        this.myList = new MutableList<Integer>();
        this.myList.addFirst(1);
        this.myList.addLast(2);
        this.myList.addLast(3);
        this.myList.addLast(4);
        this.myList.addLast(5);

    @Test

        Assert.assertEquals(this.myList.getLast().intValue(), 5);
    }

   @Test

    public void removeTesting(){

        Assert.assertEquals(emptyList.remove(0), false);

        MutableList<Integer> oneItemList = new MutableList<Integer>();
        oneItemList.addFirst(1);
        Assert.assertEquals(oneItemList.remove(1), true);
    }

    @Test
    public void sizeTesting(){
        Assert.assertEquals(this.myList.size(), 5);

        MutableList<Integer> emptyList = new MutableList<Integer>();
        Assert.assertEquals(emptyList.size(), 0);

        oneItemList.addFirst(1);
        Assert.assertEquals(oneItemList.size(), 1);

        this.myList.remove(2);
        Assert.assertEquals(this.myList.size(), 4);
    }

    @Test
    public void getNodeTesting(){

    }

    public boolean compareTwoLists(IList<Integer> l1, IList<Integer> l2) {
        if (l1.size() != l2.size()) {
            return false;
        } else {
            for (int i = 0; i < l1.size(); i++) {
                if (!(l1.get(i).equals(l2.get(i)))) {
                    return false;
                }
            }
            return true;
        }
    }

    @Test
    public void removeNodeTesting(){
        //REMOVENODE DOESN'T WORK
        MutableList<Integer> removedList = new MutableList<Integer>();
        removedList.addFirst(1);
        removedList.addLast(2);
        removedList.addLast(4);
        removedList.addLast(5);
        Node N = this.myList.getNode(2);
        this.myList.removeNode(N);
        Assert.assertEquals(this.compareTwoLists(this.myList, removedList), true);
    }
